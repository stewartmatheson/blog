require 'fileutils'
Dir.glob("*.jpg").each do |file_name|
  new_name = file_name[2..file_name.length] 
  new_name.gsub! "-", "_" 
  puts "Renaming #{file_name} => #{new_name}"
  File.rename file_name, new_name
end
